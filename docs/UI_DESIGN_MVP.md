## API Context Implementation

### Store Pattern (React/Context API)

```javascript
// YachtContext.js
import { createContext, useContext, useState, useEffect } from 'react';

const YachtContext = createContext(null);

export function YachtProvider({ children }) {
  const [yachtContext, setYachtContext] = useState({
    currentYachtId: null,
    currentYacht: null,
    effectiveRole: null
  });

  const selectYacht = (yachtId, yachtData) => {
    setYachtContext({
      currentYachtId: yachtId,
      currentYacht: yachtData,
      effectiveRole: yachtData.roleOverride || yachtData.globalRole
    });
    // Persist to localStorage
    localStorage.setItem('lastYachtId', yachtId);
  };

  const value = { yachtContext, selectYacht };
  return (
    <YachtContext.Provider value={value}>
      {children}
    </YachtContext.Provider>
  );
}

// Hook for API calls
export function useApi() {
  const { yachtContext } = useContext(YachtContext);
  
  const apiCall = async (endpoint, options = {}) => {
    const url = `/api${endpoint}?yachtId=${yachtContext.currentYachtId}`;
    return fetch(url, {
      ...options,
      headers: {
        'Authorization': `Bearer ${getToken()}`,
        'Content-Type': 'application/json',
        ...options.headers
      }
    });
  };
  
  return { apiCall };
}
```

## Quick Reference

### Component Map

| Component | Usage |
|-----------|-------|
| Primary Button | Gold, full width, primary actions |
| Secondary Button | Outlined, cancel/dismiss |
| Input | Full width, 48px, gold focus |
| Card | Elevated surface, 12px radius |
| Badge | Status indicators, pill shape |
| Modal | 640px max, centered overlay |
| Toast | Bottom right, auto-dismiss 4s |

### Color Usage

| Color | Usage |
|-------|-------|
| Gold (#D4AF37) | Primary actions, icons, accents |
| Navy (#0A0E17) | Background, text on gold |
| Success (#10B981) | Confirmations, OK status |
| Warning (#F59E0B) | Alerts, pending items |
| Error (#EF4444) | Errors, destructive actions |
| Info (#3B82F6) | Information, submitted status |

### Spacing Quick Guide

| Padding | Use Case |
|---------|----------|
| 16px | Card content, modal body |
| 24px | Card padding, section |
| 32px | Modal padding, page headers |
| 48px | Full sections, page margins |

### Typography Scale

| Size | Weight | Use |
|------|--------|-----|
| 32px | 700 | Display titles |
| 28px | 600 | Page headers |
| 24px | 600 | Card titles |
| 18px | 600 | Section headers |
| 15px | 400 | Body text |
| 13px | 400 | Labels, metadata |
| 12px | 500 | Tags, badges |

```javascript
// Toggle state
const [darkMode, setDarkMode] = useState(true);

  document.documentElement.classList.toggle('light-mode', !darkMode);
  localStorage.setItem('theme', darkMode ? 'dark' : 'light');
}, [darkMode]);
```
